      < script>
        function checkSchedule() {
          let day = document.getElementById('dayInput').value.trim();
          let message;
    
          switch (day) {
            case 'Monday':
              message = 'Start of the week meeting at 10 AM.';
              break;
            case 'Tuesday':
              message = 'Project brainstorming session at 2 PM.';
              break;
            case 'Wednesday':
              message = 'Client call at 4 PM.';
              break;
            case 'Thursday':
              message = 'Team lunch at noon.';
              break;
            case 'Friday':
              message = 'Weekly review and wrap-up at 3 PM.';
              break;
            case 'Saturday':
            case 'Sunday':
              message = 'Enjoy your weekend!';
              break;
            default:
              message = 'Invalid day. Please enter a valid day of the week.';
          }
    
          document.getElementById('output').innerText = message;
        }
      < /script>
