        function runBreakExample() {
          let output = 'Break Example Output:\n';
          for (let i = 1; i <= 10; i++) {
            if (i === 5) {
              output += 'Loop stopped at i = 5\n';
              break; // Exit the loop when i is 5
            }
            output += `Iteration ${i}\n`;
          }
          document.getElementById('output').innerText = output;
        }
    
        function runContinueExample() {
          let output = 'Continue Example Output:\n';
          for (let i = 1; i <= 10; i++) {
            if (i === 5) {
              output += 'Skipping i = 5\n';
              continue; // Skip the current iteration when i is 5
            }
            output += `Iteration ${i}\n`;
          }
          document.getElementById('output').innerText = output;
        }
      < /script>
